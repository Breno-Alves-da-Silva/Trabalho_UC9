

CREATE TABLE livro(
codigo INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
titulo VARCHAR(50),
genero VARCHAR(50),
paginas INTEGER,
sinopse VARCHAR(300),
status VARCHAR(50)
);

CREATE TABLE emprestimo(
codigo INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
dataLocacao varchar(10),
dataDevolucao varchar(10),
codigoLivro INTEGER,
codigoUsuario INTEGER,
FOREIGN KEY (codigoLivro) REFERENCES livro(codigo),
FOREIGN KEY (codigoUsuario) REFERENCES usuario(codigo)
);

CREATE TABLE usuario(
codigo INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
nome VARCHAR(50), 
cpf VARCHAR(150), 
email VARCHAR(50),
endereco VARCHAR(100),
dataNascimento VARCHAR(50)
);

CREATE TABLE autor(
codigo INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
nome VARCHAR(50),
nacionalidade VARCHAR(50)
);
