

CREATE TABLE livro(
codigo INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
titulo VARCHAR(50),
genero VARCHAR(50),
paginas INTEGER,
sinopse VARCHAR(300)
);

CREATE TABLE item(
codigo INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
status VARCHAR(50),
dataLocacao varchar(10),
dataDevolucao varchar(10)
);

CREATE TABLE usuario(
codigo INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
nome VARCHAR(50), 
cpf VARCHAR(150), 
email VARCHAR(50),
endereco VARCHAR(100),
dataNascimento VARCHAR(50)
);

CREATE TABLE autor(
codigo INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
nome VARCHAR(50),
nacionalidade VARCHAR(50)
);

CREATE TABLE possui(
codigoLivro INTEGER,
codigoAutor INTEGER,
FOREIGN KEY (codigoLivro) REFERENCES livro(codigo),
FOREIGN KEY (codigoAutor) REFERENCES autor(codigo)
);
